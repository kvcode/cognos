// Search through Cognos global objects
console.log("=== Searching for PreLoad instance ===");

// Check common Cognos namespaces
if (window.cognos) {
  console.log("✅ cognos object exists");
  console.log("cognos keys:", Object.keys(window.cognos));
}

// Check if there's a Report object
if (window.cognos && window.cognos.Report) {
  console.log("✅ cognos.Report exists");
  try {
    var report = cognos.Report.getReport("_THIS_");
    console.log("Report instance:", report);
    
    // Check if report has controls
    if (report.prompt) {
      console.log("Report has prompt object:", report.prompt);
    }
  } catch(e) {
    console.log("Could not get report:", e.message);
  }
}

// Look in DOM for the actual container
var allDivs = document.querySelectorAll('div');
console.log("Total divs on page:", allDivs.length);

// Find divs with our class names
var customDivs = document.querySelectorAll('.custom-prompt-page-container, .left-pane-container, .right-pane');
console.log("Found custom control divs:", customDivs.length);

if (customDivs.length > 0) {
  console.log("✅ Found custom control DOM elements!");
  customDivs.forEach(function(div) {
    console.log("  - Class:", div.className);
    console.log("  - ID:", div.id);
  });
}

// Check if the PreLoad constructor has getParameters
console.log("=== Checking PreLoad Constructor ===");

// Try to access through RequireJS
if (window.require) {
  try {
    // List all loaded modules
    var contexts = require.s.contexts._;
    var registry = contexts.registry;
    
    console.log("RequireJS modules loaded:");
    for (var key in registry) {
      if (key.toLowerCase().includes('preload') || key.toLowerCase().includes('custom')) {
        console.log("  Found:", key);
      }
    }
  } catch(e) {
    console.log("Cannot access RequireJS registry:", e.message);
  }
}
